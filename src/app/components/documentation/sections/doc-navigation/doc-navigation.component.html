import {{ '{' }} signal {{ '}' }} from '@angular/core';

export class NavigationComponent {{ '{' }}
  activeSection = signal('home');

  setActiveSection(section: string) {{ '{' }}
    this.activeSection.set(section);
  {{ '}' }}

    event.preventDefault();
